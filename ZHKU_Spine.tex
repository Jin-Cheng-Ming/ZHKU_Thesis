\newcommand{\makespine}{
  % 无页眉页尾 no header and footer  
  \newpage\thispagestyle{empty}
  % 行距 line space
  \linespread{1.0}
  % 字号 font property
  \heiti\zihao{4}\centering
  % 结构 structure
  \spinetitle\vfill\spineauthor\vfill\spineinst\vfill\spineyear
}%
