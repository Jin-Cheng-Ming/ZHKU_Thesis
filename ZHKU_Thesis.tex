\input{Tex/Frontinfo}

\begin{document}
  %-
  %-> 前置部分：封面、声明、摘要、目录列表、符号列表

  \frontmatter{
    \input{Tex/Frontmatter}
    \input{Tex/Prematter}
    \input{Tex/Catalogue}

    \input{Tex/Mainmatter}

  %-> 附录部分：材料、数据术语汇编

        \bibliography{Biblio/ref}

        % 使用参考文献并编排 bibliography
        \printbibliography

    \setcounter{page}{1}\pagenumbering{Roman}

    \backmatter
    \input{Tex/Backmatter}
